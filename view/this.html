Global

일반적으로 브라우저에서는 window를 의미하며, nodejs의 경우 Global Object를 의미한다. 

var counter = {
  val: 0,
  increment: function () {
    this.val += 1;
  }
};
counter.increment();
console.log(counter.val); // 1
counter['increment']();
console.log(counter.val); // 2

Function

function 영역에서의 this는 부모객체이다. 

function F (v) {
  this.val = v;
}
var f = new F(“Woohoo!”);
console.log(f.val); // Woohoo!
console.log(val); // ReferenceError

함수의 메소드인 apply, call을 이용하면 this의 값을 제어할 수 있다. 

var o = {}
var p = {}
function func(){
    switch(this){
        case o:
            document.write('o');
            break;
        case p:
            document.write('p');
            break;
        case window:
            document.write('window');
            break;          
    }
}

func(); // window
func.apply(o); // o
func.apply(p); // p
